def configure(conf)
  if path = conf['path']
    @path = path
  end
  unless @path
    raise ConfigError, "'path' parameter is required on file output"
  end

  if pos = @path.index('*')
    @path_prefix = @path[0,pos]
    @path_suffix = @path[pos+1..-1]
    conf['buffer_path'] ||= "#{@path}"
  else
    @path_prefix = @path+"."
    @path_suffix = ".log"
    conf['buffer_path'] ||= "#{@path}.*"
  end

  super

  @timef = TimeFormatter.new(@time_format, @localtime)
end

def write(chunk)
  case @compress
  when nil
    suffix = ''
  when :gz
    suffix = ".gz"
  end

  i = 0
  begin
    path = "#{@path_prefix}#{chunk.key}_#{i}#{@path_suffix}#{suffix}"
    i += 1
  end while File.exist?(path)
  FileUtils.mkdir_p File.dirname(path)

  case @compress
  when nil
    File.open(path, "a") {|f|
      chunk.write_to(f)
    }
  when :gz
    Zlib::GzipWriter.open(path) {|f|
      chunk.write_to(f)
    }
  end

  return path  # for test
end
